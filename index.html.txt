<script>

const GOOGLE_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbyaF6lznTUR99NtfxY-c0D-0Qr2hMCEnkFWusWFlC4nnwtevzshiGD2iUalqcmhsk-mqA/exec";

/* ================== รายการสินค้า ================== */

const stockItems = {

front:[
"น้ำจิ้มสุกี้","น้ำจิ้มหมูกระทะ","ซีอิ๊วขาว","ตะเกียบไม้","ทิชชู่หน้าร้าน"
],

kitchen:[
"น้ำมันงา","น้ำตาล","เห็ดหอม","ชูรส","ซีอิ๊วดำ"
],

bar:[
"น้ำกลาง","โซดา","นมข้น","หลอดสั้น","น้ำแดง"
]

};

let currentCategory="";
let employee="";

/* ================= LOGIN ================= */

function login(){
const name=document.getElementById("employeeName").value.trim();
if(!name) return alert("กรุณากรอกชื่อพนักงาน");

localStorage.setItem("employee",name);
startApp();
}

function logout(){
localStorage.removeItem("employee");
location.reload();
}

function startApp(){
employee=localStorage.getItem("employee");
if(!employee) return;

document.getElementById("login-page").classList.add("hidden");
document.getElementById("home-page").classList.remove("hidden");
document.getElementById("empDisplay").innerText=employee;
}

startApp();

/* ================= CATEGORY ================= */

function openCategory(cat){
currentCategory=cat;

document.getElementById("home-page").classList.add("hidden");
document.getElementById("category-page").classList.remove("hidden");

document.getElementById("category-title").innerText=getCategoryName();
renderItems();
}

function goHome(){
document.getElementById("home-page").classList.remove("hidden");
document.getElementById("category-page").classList.add("hidden");
document.getElementById("dashboard-page").classList.add("hidden");
}

function getCategoryName(){
if(currentCategory==="front") return "หน้าร้าน";
if(currentCategory==="kitchen") return "ในครัว";
if(currentCategory==="bar") return "บาร์น้ำ";
}

/* ================= RENDER ITEMS ================= */

function renderItems(){

const container=document.getElementById("item-list");
container.innerHTML="";

stockItems[currentCategory].forEach((item,i)=>{

container.innerHTML+=`
<div class="grid grid-cols-3 gap-3 bg-white p-3 rounded-xl shadow mb-2 fade-in">
<div>${item}</div>
<input type="number" id="qty-${i}"
class="border rounded px-2 py-1 text-center"
placeholder="0">
<input type="text" id="note-${i}"
class="border rounded px-2 py-1"
placeholder="หมายเหตุ">
</div>
`;
});

container.innerHTML+=`
<button onclick="saveStock()"
class="mt-4 bg-green-600 text-white px-6 py-3 rounded-xl w-full">
💾 บันทึกข้อมูล
</button>
`;

}

/* ================= SAVE ================= */

async function saveStock(){

let records=[];
let today=new Date().toISOString();

stockItems[currentCategory].forEach((item,i)=>{
records.push({
date:today,
employee:employee,
category:getCategoryName(),
item:item,
qty:document.getElementById(`qty-${i}`).value || "0",
note:document.getElementById(`note-${i}`).value || ""
});
});

await fetch(GOOGLE_SCRIPT_URL,{
method:"POST",
body:JSON.stringify(records),
headers:{"Content-Type":"application/json"}
});

alert("✅ บันทึกสำเร็จ");
goHome();
}

/* ================= DASHBOARD ================= */

function openDashboard(){
document.getElementById("home-page").classList.add("hidden");
document.getElementById("dashboard-page").classList.remove("hidden");
}

async function loadDashboard(){

const start=document.getElementById("startDate").value;
const end=document.getElementById("endDate").value;
if(!start||!end) return alert("กรุณาเลือกช่วงวันที่");

const res=await fetch(GOOGLE_SCRIPT_URL+"?action=getData");
const data=await res.json();

let front={}, kitchen={}, bar={};

data.forEach(r=>{
let date=r.date.split("T")[0];

if(date>=start && date<=end){

let qty=parseFloat(r.qty)||0;

if(r.category==="หน้าร้าน"){
front[r.item]=(front[r.item]||0)+qty;
}
if(r.category==="ในครัว"){
kitchen[r.item]=(kitchen[r.item]||0)+qty;
}
if(r.category==="บาร์น้ำ"){
bar[r.item]=(bar[r.item]||0)+qty;
}

}
});

renderChart("frontChart",front);
renderChart("kitchenChart",kitchen);
renderChart("barChart",bar);
}

function renderChart(id,data){

if(window[id+"Instance"]){
window[id+"Instance"].destroy();
}

window[id+"Instance"]=new Chart(document.getElementById(id),{
type:"bar",
data:{
labels:Object.keys(data),
datasets:[{
label:"จำนวนรวม",
data:Object.values(data),
borderWidth:1
}]
},
options:{
responsive:true,
plugins:{legend:{display:false}},
scales:{y:{beginAtZero:true}}
}
});
}
